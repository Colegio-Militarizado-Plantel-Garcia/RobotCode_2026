package org.firstinspires.ftc.teamcode;

import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode;
import com.qualcomm.robotcore.eventloop.opmode.TeleOp;
import com.qualcomm.robotcore.hardware.DcMotor;
import com.qualcomm.robotcore.hardware.DcMotorSimple;
import com.qualcomm.robotcore.util.Range;


@TeleOp(name = "MiPrimerPrograma", group="pruebas william")
public class MiPrimerPrograma extends LinearOpMode {

  // Declarar variables
    DcMotor motorIzquierdo;
    DcMotor motorDerecho;

    double valorpalancaizquierda, valorpalancaderecha;
    double podermotorizquierdo, podermotorderecho;

    @Override
    public void runOpMode() {

        // Inicializar las variables




       motorIzquierdo = hardwareMap.dcMotor.get("mI");
       motorDerecho = hardwareMap.dcMotor.get("mD");

       motorIzquierdo.setDirection(DcMotorSimple.Direction.REVERSE);
       motorDerecho.setDirection(DcMotorSimple.Direction.FORWARD);

       motorIzquierdo.setZeroPowerBehavior(DcMotor.ZeroPowerBehavior.BRAKE);
       motorDerecho.setZeroPowerBehavior(DcMotor.ZeroPowerBehavior.BRAKE);


       waitForStart();

        while (opModeIsActive()) {

       //Hacer las funciones que queremos que haga el robot

        // Queremos que el robot se mueva, dependiendo de lo que pongamos dentro del control.


          valorpalancaizquierda = -gamepad1.left_stick_y;
          valorpalancaderecha = gamepad1.right_stick_x;

          podermotorizquierdo = valorpalancaizquierda - valorpalancaderecha;
          podermotorderecho = valorpalancaizquierda + valorpalancaderecha;

          podermotorizquierdo = Range.clip(podermotorizquierdo,-1.0, 1.0);
          podermotorderecho = Range.clip(podermotorderecho, -1.0, 1.0);

          motorIzquierdo.setPower(podermotorizquierdo);
          motorDerecho.setPower(podermotorderecho);
        }
    }
}
