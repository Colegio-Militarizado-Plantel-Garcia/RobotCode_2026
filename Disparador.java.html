package org.firstinspires.ftc.teamcode.mechanisms;

import com.qualcomm.robotcore.eventloop.opmode.OpMode;
import com.qualcomm.robotcore.eventloop.opmode.TeleOp;
import com.qualcomm.robotcore.hardware.DcMotor;
import com.qualcomm.robotcore.hardware.DcMotorEx;
import com.qualcomm.robotcore.hardware.DcMotorSimple;
import com.qualcomm.robotcore.hardware.PIDFCoefficients;

@TeleOp(name = "Disparador TeleOp")
public class Disparador extends OpMode {

    public DcMotorEx flywheelMotor;

    // Velocidades tuneadas — ajusta estos valores con tu FlywheelTuner
    public static double HIGH_VELOCITY = 1500;
    public static double LOW_VELOCITY  = 900;

    // Coeficientes PIDF — pon aquí los valores que encontraste con el tuner
    public static double F = 0;
    public static double P = 0;

    private boolean flywheelActive = false;
    private double  curTargetVelocity = HIGH_VELOCITY;

    @Override
    public void init() {
        flywheelMotor = hardwareMap.get(DcMotorEx.class, "motorDisparador");
        flywheelMotor.setMode(DcMotor.RunMode.RUN_USING_ENCODER);
        flywheelMotor.setDirection(DcMotorSimple.Direction.REVERSE);

        applyPIDF();

        telemetry.addLine("Disparador listo");
        telemetry.update();
    }

    @Override
    public void loop() {

        // --- Encender / apagar flywheel con RB ---
        if (gamepad1.right_bumper) {
            flywheelActive = !flywheelActive;
        }

        // --- Cambiar entre velocidad alta y baja con Y ---
        if (gamepad1.y) {
            curTargetVelocity = (curTargetVelocity == HIGH_VELOCITY) ? LOW_VELOCITY : HIGH_VELOCITY;
        }

        // --- Aplicar velocidad al motor ---
        if (flywheelActive) {
            flywheelMotor.setVelocity(curTargetVelocity);
        } else {
            flywheelMotor.setVelocity(0);
        }

        // --- Telemetría ---
        double curVelocity = flywheelMotor.getVelocity();
        double error = curTargetVelocity - curVelocity;

        telemetry.addData("Flywheel", flywheelActive ? "ENCENDIDO" : "APAGADO");
        telemetry.addData("Velocidad objetivo", curTargetVelocity);
        telemetry.addData("Velocidad actual", "%.2f", curVelocity);
        telemetry.addData("Error", "%.2f", error);
        telemetry.addData("¿Listo para disparar?", Math.abs(error) < 50 ? "SÍ ✓" : "NO...");
        telemetry.update();
    }

    private void applyPIDF() {
        PIDFCoefficients pidf = new PIDFCoefficients(P, 0, 0, F);
        flywheelMotor.setPIDFCoefficients(DcMotor.RunMode.RUN_USING_ENCODER, pidf);
    }
}
