package org.firstinspires.ftc.teamcode;

import com.qualcomm.robotcore.eventloop.opmode.OpMode;
import com.qualcomm.robotcore.eventloop.opmode.TeleOp;
import com.qualcomm.robotcore.hardware.DcMotor;
import com.qualcomm.robotcore.hardware.DcMotorSimple;

@TeleOp(name = "Core Hex Motor", group = "TeleOp")
public class CoreHexMotor extends OpMode {

    DcMotor coreHex;

    boolean botonAnterior = false;
    boolean motorEncendido = false;

    @Override
    public void init() {
        coreHex = hardwareMap.get(DcMotor.class, "coreHex");


        // Configuración recomendada para Core Hex
        coreHex.setMode(DcMotor.RunMode.RUN_WITHOUT_ENCODER);
        coreHex.setZeroPowerBehavior(DcMotor.ZeroPowerBehavior.BRAKE);
        coreHex.setDirection(DcMotor.Direction.REVERSE);

        telemetry.addData("Estado", "Listo");
        telemetry.update();
    }

    @Override
    public void loop() {

        boolean botonActual = gamepad1.a;

        // Alterna encendido/apagado al presionar A
        if (botonActual && !botonAnterior) {
            motorEncendido = !motorEncendido;
        }
        botonAnterior = botonActual;

        if (motorEncendido) {
            coreHex.setPower(5.0);
        } else {
            coreHex.setPower(0.0);
        }

        telemetry.addData("Motor", motorEncendido ? "ENCENDIDO" : "APAGADO");
        telemetry.addData("Potencia", coreHex.getPower());
        telemetry.update();
    }
}
